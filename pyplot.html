# given lists of information needed for code

merge_sort_times = [0.00161889999999687, 0.003635600000009731, 0.0090655000000055, 0.020108000000007564, 0.2687474000000094, 1.6147050999999948]

# change style of pyplot
plt.style.use('bmh')

# create plot with bot sets of information on one plot and customize
plt.plot(sizes, insertion_sort_times, label='Insertion Sort Times', marker='^', color='b')
plt.plot(sizes, merge_sort_times, label='Merge Sort Times', marker='s', color='g')
